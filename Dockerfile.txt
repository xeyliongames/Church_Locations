# Use official Ubuntu as base image
FROM ubuntu:22.04

# Set environment variables to avoid interactive prompts during package install
ENV DEBIAN_FRONTEND=noninteractive

# Update and install Python and pip
RUN apt-get update && apt-get install -y \
    python3 \
    python3-pip \
    && rm -rf /var/lib/apt/lists/*

# Upgrade pip
RUN python3 -m pip install --upgrade pip

# Install required Python packages
RUN pip3 install pandas geopy haversine

# Copy the Python script into the container
COPY nearest_churches.py /app/nearest_churches.py

# Set working directory
WORKDIR /app

# Run the script by default when container starts
CMD ["python3", "nearest_churches.py"]
